    var myusers = users.find(user => user.id === comments.userId);
    var mymovies=movies.find(movie=> movie.id === comments.movieId)
    return{
        name:myusers.name, 
        movies:mymovies.name
}
}   
).reduce((prev,cur)=>{
    if(!prev[cur.name])
        {
            prev[cur.name]=cur.movies
        }
        else{
            prev[cur.name]=prev[cur.name]+" "+cur.movies;
        }
        return prev
},{})
